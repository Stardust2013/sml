38template< typename Vector3 >
39Vector3 cross( const Vector3& vector1, const Vector3& vector2 )
40{
41 // Check if either vector is not a 3-vector. If so, throw an exception.
42 if ( vector1.size( ) != 3 || vector2.size( ) != 3 )
43 {
44 throw std::runtime_error( "ERROR: Cross product can only be computed for 3-vectors!" );
45 }
46
47 Vector3 result = vector1;
48
49 // Compute components of resulting 3-vector.
50 result[ 0 ] = vector1[ 1 ] * vector2[ 2 ] - vector1[ 2 ] * vector2[ 1 ];
51 result[ 1 ] = vector1[ 2 ] * vector2[ 0 ] - vector1[ 0 ] * vector2[ 2 ];
52 result[ 2 ] = vector1[ 0 ] * vector2[ 1 ] - vector1[ 1 ] * vector2[ 0 ];
53
54 return result;
55}

77template< typename Real, typename Vector >
78Real dot( const Vector& vector1, const Vector& vector2 )
79{
80 // Check if vectors are of different length. If so, throw an exception.
81 if ( vector1.size( ) != vector2.size( ) )
82 {
83 throw std::runtime_error(
84 "ERROR: Dot product can only be computed for vectors of equal length!" );
85 }
86
87 Real result = 0.0;
88
89 // Loop through vector and compute dot product.
90 for ( unsigned int i = 0; i < vector1.size( ); i++ )
91 {
92 result += vector1[ i ] * vector2[ i ];
93 }
94
95 return result;
96}

169template< typename Real, typename Vector >
170Vector normalize( const Vector& vector )
171{
172 // Declare normalized vector.
173 Vector normalizedVector = vector;
174
175 // Compute the norm of the vector.
176 Real vectorNorm = norm< Real, Vector >( vector );
177
178 // Loop through vector and divide each component by the norm of the vector.
179 for ( unsigned int i = 0; i < vector.size( ); i++ )
180 {
181 normalizedVector[ i ] = vector[ i ] / vectorNorm;
182 }
