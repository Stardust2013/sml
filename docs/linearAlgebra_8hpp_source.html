40 Vector3 cross( const Vector3& vector1, const Vector3& vector2 )
41 {
42  // Check if either vector is not a 3-vector. If so, throw an exception.
43  if ( vector1.size( ) != 3 || vector2.size( ) != 3 )
44  {
45  throw std::runtime_error( "ERROR: Cross product can only be computed for 3-vectors!" );
46  }
47 
48  Vector3 result = vector1;
49 
50  // Compute components of resulting 3-vector.
51  result[ 0 ] = vector1[ 1 ] * vector2[ 2 ] - vector1[ 2 ] * vector2[ 1 ];
52  result[ 1 ] = vector1[ 2 ] * vector2[ 0 ] - vector1[ 0 ] * vector2[ 2 ];
53  result[ 2 ] = vector1[ 0 ] * vector2[ 1 ] - vector1[ 1 ] * vector2[ 0 ];
54
